Example

  1. // Most simple scenario 
  2. Social.Bitly bit = Social.Bitly("username", "R_someApiKey876hd723");  
  3. string longurl = "http://some-freaking.long/url/man/that/never/ends/";  
  4. Social.Bitly.Response resp = bit->shorten(longurl);  
  5. werror("%O\n", resp);  
  6.   
  7. if (resp->success())  
  8.   write("Short url: %s\n", resp->result()->nodeKeyVal->shortUrl);  
  9. else  
  10.  write("Error: %s\n", resp->error_message());
